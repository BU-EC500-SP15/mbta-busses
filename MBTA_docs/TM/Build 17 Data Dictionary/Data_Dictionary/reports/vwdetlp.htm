Definition
Note
DDL Code CREATE VIEW v_veh_sched_adherence
(CALENDAR_ID, ROUTE_ID, ROUTE_DIRECTION_ID, GEO_NODE_ID, OPERATOR_ID, BLOCK_ID, TP_ADHERENCE, SCHEDULED_TIME, ACT_ARRIVAL_TIME, WAIVER_EARLY_ALLOWED, WAIVER_LATE_ALLOWED, ACT_DEPARTURE_TIME, ACTUAL_VEHICLE_ID, OVERLOAD_FLAG, ACTUAL_PULLOUT_TIME, ACTUAL_PULLIN_TIME) AS
SELECT
   TPC.CALENDAR_ID
, TPC.ROUTE_ID
, TPC.ROUTE_DIRECTION_ID
, TPC.GEO_NODE_ID
, DR.OPERATOR_ID
, DP.BLOCK_ID
, ((tpc.scheduled_time - tpc.act_arrival_time)/60)
, TPC.SCHEDULED_TIME
, TPC.ACT_ARRIVAL_TIME
, (CASE WHEN saw.early_allowed_flag = 1 THEN 'Y'
      ELSE 'N' END )
, (CASE WHEN saw.late_allowed_flag = 1 THEN 'Y'
      ELSE 'N' END )
, TPC.ACT_DEPARTURE_TIME
, DP.ACTUAL_VEHICLE_ID
, DP.OVERLOAD_FLAG
, DP.ACTUAL_PULLOUT_TIME
, DP.ACTUAL_PULLIN_TIME
FROM
   TIME_POINT_CROSSING TPC
     JOIN
   DAILY_TRIP DT
      ON tpc.daily_trip_id = dt.daily_trip_id
     JOIN
   DAILY_PULLOUT DP
      ON dt.pullout_id = dp.pullout_id
     LEFT OUTER JOIN
   SCHED_ADHERE_WAIVER SAW
      ON tpc.waiver_id = saw.waiver_id
     LEFT OUTER JOIN
   DAILY_WORK_PIECE   DWP
      ON dt.daily_work_piece_id = dwp.daily_work_piece_id
     LEFT OUTER JOIN
   DAILY_RUN   DR
      ON dwp.daily_run_id = dr.daily_run_id
